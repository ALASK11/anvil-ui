# ============================================================
# Cloud SQL — IAM Database Authentication
# ============================================================
# Enable IAM auth on the Cloud SQL instance so the Cloud Run SA can
# authenticate to Postgres without a password. Triggers an instance
# restart. WARNING: --database-flags REPLACES the full flag set; if
# other flags exist, list them all here too. Verify first with:
#   gcloud sql instances describe anvil-intelligence-db \
#     --project=anvil-private \
#     --format="value(settings.databaseFlags)"

gcloud sql instances patch anvil-intelligence-db \
  --project=anvil-private \
  --database-flags=cloudsql.iam_authentication=on

# Project-level roles for the Next.js Cloud Run SA.
# - cloudsql.client       → establish the TLS tunnel
# - cloudsql.instanceUser → IAM-based Postgres login

gcloud projects add-iam-policy-binding anvil-private \
  --member="serviceAccount:169801273048-compute@developer.gserviceaccount.com" \
  --role="roles/cloudsql.client"

gcloud projects add-iam-policy-binding anvil-private \
  --member="serviceAccount:169801273048-compute@developer.gserviceaccount.com" \
  --role="roles/cloudsql.instanceUser"

# Create the IAM Postgres user. Pass the SA email WITHOUT the
# .gserviceaccount.com suffix — Cloud SQL drops it to fit Postgres's
# 63-char identifier limit (so the actual Postgres role becomes
# "169801273048-compute@developer").

gcloud sql users create 169801273048-compute@developer \
  --instance=anvil-intelligence-db \
  --project=anvil-private \
  --type=cloud_iam_service_account

# Cloud Run resource sizing (private-nextjs-site)
# ============================================================

# 5 GiB requires ≥ 2 vCPU per Cloud Run's memory/CPU pairing rules.

  --region=us-central1 \
  --project=anvil-private \
  --memory=5Gi \
  --cpu=2
